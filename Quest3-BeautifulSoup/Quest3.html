        else:
            percentchange = float(re.sub(r'\D.\D(?s).*</span></td>', '', re.sub(r'<td><span class=".*">', '', str(tableNode[i-1]))))

        percentlist.append(percentchange)

# Utility function
def runtime(f):
    import timeit
    start = timeit.default_timer()
    f
    end = timeit.default_timer()
    return(end - start) 

# My Function
def wordsInStringToDictWordCount(istr):
    
    return_dict = {}
    
    if len(istr) > 1:      
        temp_list = istr.split()
        
        for i in temp_list:
            if i in return_dict:
                return_dict[i] += 1
            else:
                return_dict[i] = 1
                
    return return_dict

# Counter class
def beatTheCounter(istr):
    cnt = Counter()
    temp_list = istr.split()
    
    for i in temp_list:
        cnt[i] += 1

    return cnt
